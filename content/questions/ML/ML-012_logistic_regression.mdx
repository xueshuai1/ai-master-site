## 参考答案

### 逻辑回归原理

**核心思想：** 通过 Sigmoid 函数将线性回归的输出映射到 (0, 1) 区间，表示属于正类的概率。

**假设函数：**
```
h_θ(x) = σ(θ^T x) = 1 / (1 + e^(-θ^T x))
```

**决策边界：**
```
if h_θ(x) ≥ 0.5 → 预测 1
if h_θ(x) < 0.5 → 预测 0
```

```python
import numpy as np
import matplotlib.pyplot as plt
from sklearn.linear_model import LogisticRegression
from sklearn.datasets import make_classification

# Sigmoid 函数
def sigmoid(z):
    return 1 / (1 + np.exp(-z))

# 可视化 Sigmoid
z = np.linspace(-10, 10, 100)
plt.figure(figsize=(10, 4))

plt.subplot(1, 2, 1)
plt.plot(z, sigmoid(z), linewidth=2)
plt.axhline(y=0.5, color='r', linestyle='--', label='Decision boundary')
plt.axvline(x=0, color='g', linestyle='--', alpha=0.5)
plt.xlabel('z = θ^T x')
plt.ylabel('σ(z)')
plt.title('Sigmoid Function')
plt.legend()
plt.grid(True, alpha=0.3)

# 生成数据并训练
X, y = make_classification(n_samples=200, n_features=2, n_informative=2, 
                           n_redundant=0, random_state=42)

clf = LogisticRegression()
clf.fit(X, y)

# 决策边界
xx, yy = np.meshgrid(np.linspace(X[:, 0].min() - 1, X[:, 0].max() + 1, 200),
                     np.linspace(X[:, 1].min() - 1, X[:, 1].max() + 1, 200))
Z = clf.predict_proba(np.c_[xx.ravel(), yy.ravel()])[:, 1]
Z = Z.reshape(xx.shape)

plt.subplot(1, 2, 2)
plt.contourf(xx, yy, Z, levels=50, cmap='RdBu', alpha=0.3)
plt.scatter(X[:, 0], X[:, 1], c=y, cmap='viridis', s=50, edgecolors='k', alpha=0.6)
plt.xlabel('Feature 1')
plt.ylabel('Feature 2')
plt.title('Logistic Regression Decision Boundary')
plt.grid(True, alpha=0.3)

plt.tight_layout()
plt.show()
```

### 损失函数推导

**为什么不用 MSE？**
```python
# 逻辑回归用 MSE 会导致非凸优化问题
# 存在多个局部最优

# 使用交叉熵损失（对数损失）
def log_loss_manual(y_true, y_proba):
    """手动计算对数损失"""
    epsilon = 1e-15  # 防止 log(0)
    y_proba = np.clip(y_proba, epsilon, 1 - epsilon)
    return -np.mean(y_true * np.log(y_proba) + (1 - y_true) * np.log(1 - y_proba))

# 交叉熵损失公式：
# L = -[y × log(p) + (1-y) × log(1-p)]
# 其中 y ∈ {0, 1}, p = h_θ(x)

# 梯度：
# ∂L/∂θ = (h_θ(x) - y) × x
```

### 正则化逻辑回归

```python
from sklearn.linear_model import LogisticRegression

# L2 正则化（默认）
clf_l2 = LogisticRegression(
    penalty='l2',
    C=1.0,          # 正则化强度的倒数
    solver='lbfgs',
    max_iter=1000
)

# L1 正则化（特征选择）
clf_l1 = LogisticRegression(
    penalty='l1',
    C=1.0,
    solver='liblinear',  # L1 需要 liblinear 或 saga
    max_iter=1000
)

# ElasticNet
clf_en = LogisticRegression(
    penalty='elasticnet',
    C=1.0,
    l1_ratio=0.5,     # L1 比例
    solver='saga',
    max_iter=1000
)

# 无正则化
clf_none = LogisticRegression(
    penalty=None,
    solver='lbfgs',
    max_iter=1000
)

# C 的影响
# C 大 → 弱正则化 → 可能过拟合
# C 小 → 强正则化 → 可能欠拟合
```

### 多分类逻辑回归

```python
from sklearn.datasets import load_iris
from sklearn.linear_model import LogisticRegression
from sklearn.model_selection import cross_val_score

# 加载数据
iris = load_iris()
X, y = iris.data, iris.target

# 多分类策略
# 1. One-vs-Rest (OvR) - 默认
clf_ovr = LogisticRegression(multi_class='ovr', solver='lbfgs')
scores_ovr = cross_val_score(clf_ovr, X, y, cv=5)
print(f"OvR Accuracy: {scores_ovr.mean():.4f}")

# 2. Multinomial (Softmax)
clf_multi = LogisticRegression(multi_class='multinomial', solver='lbfgs')
scores_multi = cross_val_score(clf_multi, X, y, cv=5)
print(f"Multinomial Accuracy: {scores_multi.mean():.4f}")

# Softmax 函数
def softmax(z):
    """Softmax 函数（多分类）"""
    exp_z = np.exp(z - np.max(z, axis=1, keepdims=True))  # 数值稳定
    return exp_z / np.sum(exp_z, axis=1, keepdims=True)

# 多分类损失（交叉熵）
# L = -Σ y_i × log(p_i)
# 其中 y 是 one-hot 编码
```

### 逻辑回归 vs 线性回归

| 特性 | 线性回归 | 逻辑回归 |
|------|---------|---------|
| **任务类型** | 回归 | 分类 |
| **输出** | 连续值 | 概率 (0-1) |
| **假设函数** | h(x) = θ^T x | h(x) = σ(θ^T x) |
| **损失函数** | MSE | 交叉熵 |
| **假设** | 正态分布 | 伯努利分布 |
| **评估指标** | R², MSE | Accuracy, AUC |

### 追问 1: 为什么逻辑回归用交叉熵而不是 MSE？

**问题：** 从优化角度解释损失函数选择。

**答案：**

```python
# 交叉熵 vs MSE 的梯度对比

def mse_gradient(y_true, y_pred):
    """MSE 梯度"""
    return 2 * (y_pred - y_true)

def cross_entropy_gradient(y_true, y_pred):
    """交叉熵梯度（经过 sigmoid）"""
    return y_pred - y_true  # 简洁！

# MSE 问题：
# 1. 非凸（多个局部最优）
# 2. 梯度消失（预测接近 0 或 1 时）
# 3. 对异常值敏感

# 交叉熵优势：
# 1. 凸函数（全局最优）
# 2. 梯度合理（误差大梯度大）
# 3. 来自极大似然估计

# 可视化
z = np.linspace(-10, 10, 100)
y = 1  # 真实标签

p_sigmoid = sigmoid(z)
loss_mse = (p_sigmoid - y) ** 2
loss_ce = -(y * np.log(p_sigmoid + 1e-15))

plt.figure(figsize=(12, 4))

plt.subplot(1, 2, 1)
plt.plot(z, loss_mse, label='MSE')
plt.xlabel('z = θ^T x')
plt.ylabel('Loss')
plt.title('MSE Loss (Non-convex for Logistic)')
plt.legend()
plt.grid(True, alpha=0.3)

plt.subplot(1, 2, 2)
plt.plot(z, loss_ce, color='orange', label='Cross-Entropy')
plt.xlabel('z = θ^T x')
plt.ylabel('Loss')
plt.title('Cross-Entropy Loss (Convex)')
plt.legend()
plt.grid(True, alpha=0.3)

plt.tight_layout()
plt.show()
```

### 追问 2: 逻辑回归能处理非线性关系吗？

**问题：** 如何扩展逻辑回归处理非线性？

**答案：**

```python
from sklearn.preprocessing import PolynomialFeatures
from sklearn.pipeline import Pipeline

# 方法 1: 多项式特征
poly_clf = Pipeline([
    ('poly', PolynomialFeatures(degree=2, include_bias=False)),
    ('clf', LogisticRegression())
])
poly_clf.fit(X, y)

# 方法 2: 核逻辑回归
from sklearn.kernel_approximation import RBFSampler

rbf_kernel = Pipeline([
    ('rbf', RBFSampler(gamma=1.0, n_components=100)),
    ('clf', LogisticRegression())
])
rbf_kernel.fit(X, y)

# 方法 3: 分箱离散化
from sklearn.preprocessing import KBinsDiscretizer

binned_clf = Pipeline([
    ('bin', KBinsDiscretizer(n_bins=5, encode='onehot')),
    ('clf', LogisticRegression())
])
binned_clf.fit(X, y)

# 方法 4: 交互特征
from itertools import combinations

def add_interactions(X):
    """添加交互特征"""
    interactions = []
    for i, j in combinations(range(X.shape[1]), 2):
        interactions.append((X[:, i] * X[:, j]).reshape(-1, 1))
    return np.hstack([X] + interactions)

X_interact = add_interactions(X)
```

## 深入理解

**逻辑回归的本质：** 广义线性模型，伯努利分布的极大似然估计

**关键要点：**
1. Sigmoid 将线性输出映射为概率
2. 交叉熵来自极大似然
3. 正则化防止过拟合
4. 可扩展为多分类（Softmax）
