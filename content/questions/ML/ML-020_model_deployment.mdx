```python
# 1. REST API (Flask/FastAPI)
from flask import Flask, request, jsonify
import pickle

app = Flask(__name__)
model = pickle.load(open('model.pkl', 'rb'))

@app.route('/predict', methods=['POST'])
def predict():
    data = request.json
    features = np.array(data['features']).reshape(1, -1)
    prediction = model.predict(features)[0]
    probability = model.predict_proba(features)[0]
    return jsonify({
        'prediction': int(prediction),
        'probability': float(probability)
    })
```

```python
# 数据漂移检测
from sklearn.ensemble import RandomForestClassifier
from sklearn.model_selection import train_test_split

def detect_data_drift(X_train, X_current):
    """检测数据漂移"""
    # 训练数据 vs 当前数据
    X_combined = np.vstack([X_train, X_current])
    y_combined = np.hstack([
        np.zeros(len(X_train)),  # 训练数据标签 0
        np.ones(len(X_current))   # 当前数据标签 1
    ])
    
    # 训练分类器区分
    clf = RandomForestClassifier()
    clf.fit(X_combined, y_combined)
    
    # 如果能区分，说明有漂移
    score = clf.score(X_combined, y_combined)
    drift_detected = score > 0.7
    
    # 特征重要性显示哪些特征漂移
    importances = clf.feature_importances_
    
    return drift_detected, importances
```

### A/B 测试

```python
# 对比新旧模型
from scipy import stats

def ab_test_model(y_true, y_pred_old, y_pred_new):
    """A/B 测试对比模型"""
    # 计算各模型指标
    from sklearn.metrics import accuracy_score
    
    acc_old = accuracy_score(y_true, y_pred_old)
    acc_new = accuracy_score(y_true, y_pred_new)
    
    # 统计检验
    t_stat, p_value = stats.ttest_rel(
        (y_pred_old == y_true).astype(int),
        (y_pred_new == y_true).astype(int)
    )
    
    significant = p_value < 0.05
    better = acc_new > acc_old
    
    return {
        'acc_old': acc_old,
        'acc_new': acc_new,
        'improvement': acc_new - acc_old,
        'p_value': p_value,
        'significant': significant,
        'better': better
    }

# 结果解读
# p < 0.05 且 improvement > 0: 新模型显著更好
# p >= 0.05: 差异不显著
```
