## 题目描述
请详细解释 PCA (主成分分析) 的原理、计算步骤和应用场景。如何选择主成分的数量？PCA 有什么局限性？与 LDA 有什么区别？

## 参考答案

### PCA 核心思想

**目标：** 将高维数据投影到低维空间，同时保留尽可能多的信息（方差）。

**核心思路：**
1. 找到数据方差最大的方向（第一主成分）
2. 找到与之前方向正交且方差次大的方向（第二主成分）
3. 重复以上步骤

### 数学原理

```python
import numpy as np
import matplotlib.pyplot as plt
from sklearn.decomposition import PCA
from sklearn.preprocessing import StandardScaler

# 1. 标准化数据（PCA 对尺度敏感）
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# 2. 计算协方差矩阵
cov_matrix = np.cov(X_scaled.T)
print(f"Covariance matrix shape: {cov_matrix.shape}")

# 3. 计算特征值和特征向量
eigenvalues, eigenvectors = np.linalg.eig(cov_matrix)
print(f"Eigenvalues: {eigenvalues}")

# 4. 计算方差解释率
variance_explained = eigenvalues / np.sum(eigenvalues)
cumulative_variance = np.cumsum(variance_explained)

print(f"\nVariance explained by each PC:")
for i, (ev, ve) in enumerate(zip(eigenvalues, variance_explained), 1):
    print(f"  PC{i}: {ve:.4f} ({ve*100:.2f}%)")

print(f"\nCumulative variance:")
for i, cv in enumerate(cumulative_variance, 1):
    print(f"  PC{i}: {cv:.4f} ({cv*100:.2f}%)")

# 5. 使用 sklearn 实现 PCA
pca = PCA(n_components=0.95)  # 保留 95% 方差
X_pca = pca.fit_transform(X_scaled)

print(f"\nOriginal dimensions: {X.shape[1]}")
print(f"After PCA: {X_pca.shape[1]}")
print(f"Variance explained: {sum(pca.explained_variance_ratio_):.4f}")
```

### 可视化 PCA

```python
# 可视化主成分
def plot_pca_components(pca, feature_names):
    """绘制主成分载荷图"""
    plt.figure(figsize=(10, 6))
    
    for i, pc in enumerate(pca.components_[:2]):  # 前两个主成分
        plt.barh(feature_names, pc, alpha=0.6, label=f'PC{i+1}')
        plt.axvline(x=0, color='black', linestyle='--', alpha=0.3)
    
    plt.xlabel('Loading')
    plt.title('PCA Component Loadings')
    plt.legend()
    plt.tight_layout()
    plt.show()

# 可视化降维效果

    """绘制 PCA 降维后的散点图"""
    plt.figure(figsize=(8, 6))
    scatter = plt.scatter(X_pca[:, 0], X_pca[:, 1], c=y, cmap='viridis', alpha=0.6)

    plt.ylabel(f'PC2 ({pca.explained_variance_ratio_[1]:.2%})')
    plt.title('PCA Projection')
    plt.colorbar(scatter, label='Class')
    plt.grid(True, alpha=0.3)
    plt.tight_layout()
    plt.show()
```

### 选择主成分数量

```python
# 方法 1: 累积方差解释率
pca = PCA()
pca.fit(X_scaled)

cumulative_variance = np.cumsum(pca.explained_variance_ratio_)
n_components_95 = np.argmax(cumulative_variance >= 0.95) + 1
n_components_99 = np.argmax(cumulative_variance >= 0.99) + 1

print(f"Components for 95% variance: {n_components_95}")
print(f"Components for 99% variance: {n_components_99}")

# 可视化
plt.figure(figsize=(10, 4))

plt.subplot(1, 2, 1)
plt.plot(range(1, len(cumulative_variance)+1), cumulative_variance, 'o-')
plt.axhline(y=0.95, color='r', linestyle='--', label='95% threshold')
plt.axhline(y=0.99, color='g', linestyle='--', label='99% threshold')
plt.xlabel('Number of Components')
plt.ylabel('Cumulative Variance Explained')
plt.title('Cumulative Variance vs Components')
plt.legend()
plt.grid(True, alpha=0.3)

plt.subplot(1, 2, 2)
plt.bar(range(1, len(pca.explained_variance_ratio_)+1), pca.explained_variance_ratio_)
plt.xlabel('Principal Component')
plt.ylabel('Variance Explained')
plt.title('Scree Plot')
plt.grid(True, alpha=0.3, axis='y')

plt.tight_layout()
plt.show()

# 方法 2: 肘部法则
# 在 Scree Plot 中找到"肘部"点

# 方法 3: 交叉验证
from sklearn.model_selection import cross_val_score
from sklearn.ensemble import RandomForestClassifier

scores = []
for n_comp in range(1, X_scaled.shape[1] + 1):
    pca = PCA(n_components=n_comp)
    X_transformed = pca.fit_transform(X_scaled)
    
    model = RandomForestClassifier(n_estimators=50, random_state=42)
    score = cross_val_score(model, X_transformed, y, cv=5, scoring='accuracy').mean()
    scores.append(score)

best_n = np.argmax(scores) + 1
print(f"Best number of components (CV): {best_n}")
print(f"Best CV accuracy: {max(scores):.4f}")
```

### PCA vs LDA

| 特性 | PCA | LDA |
|------|-----|-----|
| **类型** | 无监督 | 有监督 |
| **目标** | 最大化方差 | 最大化类间分离 |
| **使用标签** | 否 | 是 |
| **最大维度** | min(n_samples, n_features) | n_classes - 1 |
| **适用场景** | 降维、可视化、去噪 | 分类前的特征提取 |

```python
from sklearn.discriminant_analysis import LinearDiscriminantAnalysis

# LDA
lda = LDA(n_components=1)  # 最多 n_classes - 1
X_lda = lda.fit_transform(X_scaled, y)

# PCA
pca = PCA(n_components=2)
X_pca = pca.fit_transform(X_scaled)

# 对比可视化
fig, axes = plt.subplots(1, 2, figsize=(12, 5))

axes[0].scatter(X_pca[:, 0], X_pca[:, 1], c=y, alpha=0.6)
axes[0].set_title('PCA (unsupervised)')
axes[0].set_xlabel('PC1')
axes[0].set_ylabel('PC2')

axes[1].scatter(X_lda[:, 0], np.zeros_like(X_lda[:, 0]), c=y, alpha=0.6)
axes[1].set_title('LDA (supervised)')
axes[1].set_xlabel('LD1')
axes[1].set_yticks([])

plt.tight_layout()
plt.show()
```

### 追问 1: 为什么 PCA 前要标准化？

**问题：** 不标准化会有什么问题？

**答案：**

```python
# 示例：特征尺度不同
np.random.seed(42)
X_unscaled = np.random.randn(100, 2)
X_unscaled[:, 0] *= 100  # 特征 1：尺度大
X_unscaled[:, 1] *= 1    # 特征 2：尺度小

# 不标准化
pca_unscaled = PCA(n_components=1)
X_pca_unscaled = pca_unscaled.fit_transform(X_unscaled)

print(f"Without scaling:")
print(f"  PC1 variance explained: {pca_unscaled.explained_variance_ratio_[0]:.4f}")
print(f"  PC1 loading: {pca_unscaled.components_[0]}")
# 结果：PC1 几乎完全由特征 1 主导

# 标准化后
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X_unscaled)

pca_scaled = PCA(n_components=1)
X_pca_scaled = pca_scaled.fit_transform(X_scaled)

print(f"\nWith scaling:")
print(f"  PC1 variance explained: {pca_scaled.explained_variance_ratio_[0]:.4f}")
print(f"  PC1 loading: {pca_scaled.components_[0]}")
# 结果：两个特征都有贡献
```

### 追问 2: PCA 的 SVD 实现是什么？

**问题：** PCA 和 SVD 有什么关系？

**答案：**

```python
# PCA 可以通过 SVD 实现
# X = U Σ V^T

# 方法 1: 特征分解（传统）
cov_matrix = np.cov(X_scaled.T)
eigenvalues, eigenvectors = np.linalg.eig(cov_matrix)

# 方法 2: SVD（数值更稳定）
U, S, Vt = np.linalg.svd(X_scaled, full_matrices=False)

# SVD 与 PCA 的关系
# 主成分 = Vt (右奇异向量)
# 特征值 = S² / (n-1)
# 投影 = U × S

eigenvalues_svd = S**2 / (X_scaled.shape[0] - 1)
variance_explained_svd = eigenvalues_svd / np.sum(eigenvalues_svd)

print(f"Eigenvalues from SVD: {eigenvalues_svd[:5]}")
print(f"Variance explained: {variance_explained_svd[:5]}")

# sklearn 的 PCA 默认使用 SVD
pca = PCA(n_components=2, svd_solver='full')  # 或 'arpack', 'randomized'
```

### 追问 3: 如何处理类别不平衡时的 PCA？

**问题：** 类别不平衡会影响 PCA 吗？

**答案：**

```python
# PCA 是无监督的，不受类别标签影响
# 但样本分布会影响方差计算

# 如果少数类样本太少，其方差可能被忽略

# 解决方案：
# 1. 分层采样确保代表性
# 2. 对各类别分别计算 PCA
# 3. 使用 LDA（有监督）

from sklearn.utils import resample

# 分层采样
X_minority = X[y == 1]
X_majority = X[y == 0]

# 过采样少数类
X_minority_upsampled = resample(X_minority, 
                                 replace=True, 
                                 n_samples=len(X_majority),
                                 random_state=42)

X_balanced = np.vstack([X_majority, X_minority_upsampled])

# 在平衡数据上应用 PCA
pca = PCA(n_components=0.95)
X_pca = pca.fit_transform(StandardScaler().fit_transform(X_balanced))
```

## 深入理解

**PCA 的本质：** 正交变换，将相关变量转换为不相关的主成分。

**关键要点：**
1. 标准化是必须的
2. 主成分数量通过累积方差选择
3. PCA 是无监督的，不考虑标签
4. 适合线性降维，非线性用 Kernel PCA 或 t-SNE
