### 协同过滤

```python
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity

# 用户 - 物品矩阵
# 行：用户，列：物品，值：评分
R = np.array([
    [5, 3, 0, 1],
    [4, 0, 0, 1],
    [1, 1, 0, 5],
    [1, 0, 0, 4],
    [0, 1, 5, 4]
])

# User-based CF
user_similarity = cosine_similarity(R)
print("User similarity matrix:")
print(user_similarity)

# 预测用户 0 对物品 2 的评分
# 用相似用户的评分加权平均
def predict_user_based(user_idx, item_idx, R, user_sim, k=3):
    # 找到对物品评分的相似用户
    rated_users = np.where(R[:, item_idx] > 0)[0]
    rated_users = rated_users[rated_users != user_idx]
    
    if len(rated_users) == 0:
        return 0
    
    # 选择最相似的 k 个用户
    similarities = user_sim[user_idx, rated_users]
    top_k_idx = np.argsort(similarities)[-k:]
    top_users = rated_users[top_k_idx]
    
    # 加权平均
    numerator = np.sum(similarities[top_k_idx] * R[top_users, item_idx])
    denominator = np.sum(np.abs(similarities[top_k_idx]))
    
    return numerator / denominator if denominator > 0 else 0

pred = predict_user_based(0, 2, R, user_similarity)
print(f"\nPredicted rating for user 0, item 2: {pred:.2f}")

# Item-based CF
item_similarity = cosine_similarity(R.T)
```

```python
def matrix_factorization(R, n_factors=10, n_epochs=100, lr=0.01, reg=0.02):
    n_users, n_items = R.shape
    
    # 随机初始化
    U = np.random.normal(0, 0.1, (n_users, n_factors))
    V = np.random.normal(0, 0.1, (n_items, n_factors))
    
    # 掩码（已知评分）
    mask = R > 0
    
    for epoch in range(n_epochs):
        # 预测
        pred = U @ V.T
        
        # 误差
        error = (R - pred) * mask
        
        # 梯度下降
        U += lr * (error @ V - reg * U)
        V += lr * (error.T @ U - reg * V)
        
        # 计算 RMSE
        rmse = np.sqrt(np.sum(error**2) / np.sum(mask))
        if (epoch + 1) % 20 == 0:
            print(f"Epoch {epoch+1}, RMSE: {rmse:.4f}")
    
    return U, V

U, V = matrix_factorization(R, n_factors=5, n_epochs=100)
```
